31  {
32  auto pos = std::remove(data.begin(), data.end(), 0);
33  data.erase(pos, data.end());
34  double sum = std::accumulate(data.begin(), data.end(), 0.0);
35  mean = sum / data.size();
36 
37  double sq_sum = std::inner_product(data.begin(), data.end(), data.begin(), 0.0);
38  stdev = std::sqrt(sq_sum / data.size() - mean * mean);
39  size_t s =data.size();
40  count_boxes = 1 + log2(data.size());
41  chi_squared theoretical_dist(count_boxes - 1);
42  quantile_value = quantile(theoretical_dist, 1-alpha);
43 
44  double expected_frequency = double(data.size()) / count_boxes;
45  statistics_value = 0;
46  for(int j=0; j < count_boxes; ++j)
47  {
48  std::size_t count_drops_in_box = std::count_if(data.begin(), data.end(),
49  [&](std::uint64_t x)
50  {
51  double size_of_box = double(*std::max_element(data.begin(), data.end())) / count_boxes;
52  return size_of_box*j <= x && x<size_of_box*(j+1);});
53  double delta_freq = count_drops_in_box - expected_frequency;
54  delta_freq *= delta_freq;
55 
56  statistics_value+=delta_freq/expected_frequency;
57  }
58  is_uniform = statistics_value < quantile_value;
59  std::cout << *this;
60  }
